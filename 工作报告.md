```c
                current_page = page;
            }
            input_idx = 0;
        } else if (key == 'q') {

        }
    }
}
```

### 迭代 3：模糊匹配筛选
**目标：** 支持按协议、IP、端口筛选，IP 支持通配符 `*` 模糊匹配。

**关键实现：**
1. 定义筛选配置结构：
```c
typedef struct {
    uint8_t protocol;  // 6=TCP, 17=UDP, 1=ICMP, 0=不限制
    char src_ip[16];   // 支持 "192.168.*"
    char dst_ip[16];
    uint16_t src_port;
    uint16_t dst_port;
    uint16_t port;     // 任意方向端口
} FilterOptions;
```

2. 通配符匹配函数：
```c
static int wildcard_match(const char *pattern, const char *str) {
    while (*pattern && *str) {
        if (*pattern == '*') {
            pattern++;
            if (*pattern == '\0') return 1;  // * 在末尾匹配任意
            // 查找 pattern 后面的内容在 str 中的位置
            while (*str) {
                if (wildcard_match(pattern, str)) return 1;
                str++;
            }
            return 0;
        } else if (*pattern == *str) {
            pattern++;
            str++;
        } else {
            return 0;
        }
    }
    return (*pattern == '\0' && *str == '\0') || (*pattern == '*');
}
```

3. 筛选判断函数：
```c
int matches_filter(const PacketInfo *info, const FilterOptions *filter) {
    // 协议筛选
    if (filter->protocol != 0 && info->protocol != filter->protocol) return 0;
    
    // 源 IP 筛选（支持通配符）
    if (filter->src_ip[0] != '\0' && !wildcard_match(filter->src_ip, info->src_ip)) return 0;
    
    // 目的 IP 筛选
    if (filter->dst_ip[0] != '\0' && !wildcard_match(filter->dst_ip, info->dst_ip)) return 0;
    
    // 端口筛选...
    // ...
    
    return 1;
}
```

4. 在扫描时就应用筛选，只缓存匹配的包：
```c
// 在 scan_callback 中
if (ctx->filter && !matches_filter(&info, ctx->filter)) {
    packet_counter++;
    return;  // 不匹配就跳过，不缓存
}
// 匹配才缓存...
```

### 迭代 4：流量概览仪表盘
**目标：** 在进入列表前，先显示一个流量统计概览。

**关键实现：**
1. 定义统计结构：
```c
typedef struct {
    uint64_t total_bytes;
    int tcp_count;
    int udp_count;
    int icmp_count;
    int other_count;
    long first_sec;
    long first_usec;
    long last_sec;
    long last_usec;
    // 常用端口统计
    int port_80;
    int port_443;
    int port_53;
    int port_22;
} TrafficStats;
```

2. 在扫描时更新统计：
```c
static void update_stats(TrafficStats *stats, const PacketInfo *info, long sec, long usec) {
    stats->total_bytes += info->length;
    
    if (info->protocol == 6) stats->tcp_count++;
    else if (info->protocol == 17) stats->udp_count++;
    else if (info->protocol == 1) stats->icmp_count++;
    else stats->other_count++;
    
    // 记录时间跨度
    if (stats->first_sec == -1 || sec < stats->first_sec) {
        stats->first_sec = sec;
        stats->first_usec = usec;
    }
    if (sec > stats->last_sec || (sec == stats->last_sec && usec > stats->last_usec)) {
        stats->last_sec = sec;
        stats->last_usec = usec;
    }
    
    // 常用端口统计
    if (info->src_port == 80 || info->dst_port == 80) stats->port_80++;
    if (info->src_port == 443 || info->dst_port == 443) stats->port_443++;
    if (info->src_port == 53 || info->dst_port == 53) stats->port_53++;
    if (info->src_port == 22 || info->dst_port == 22) stats->port_22++;
}
```

3. 显示仪表盘函数：
```c
void pcap_show_stats(PcapContext *ctx_) {
```

### 迭代 5：数据包标记功能
**目标：** 支持标记感兴趣的数据包，在列表中高亮显示。

**关键实现：**
1. 增加标记数组：
```c
typedef struct {
    // ...
    int *marked;  // 标记数组：1 表示被标记，0 表示未标记
} PcapContextInternal;

// 扫描完成后分配标记数组
ctx->marked = (int*)calloc(ctx->count, sizeof(int));
```

2. 增加标记操作函数：
```c
void pcap_toggle_mark(PcapContext *ctx_, int index) {
    PcapContextInternal *ctx = (PcapContextInternal*)ctx_;
    if (!ctx || !ctx->marked || index < 0 || index >= ctx->count) return;
    ctx->marked[index] = !ctx->marked[index];
}
```

3. 在显示时增加 "M" 列，用 `*` 表示被标记：

4. 交互模式增加标记子模式：
```c

char mark_input[8];

    in_mark_mode = 1;  // 进入标记模式
    mark_input_idx = 0;
    input_idx = 0;

    if (key == '\r' || key == '\n') {
        if (mark_input_idx > 0) {
            mark_input[mark_input_idx] = '\0';
            int line_num = atoi(mark_input);
            if (line_num >= 1 && line_num <= count) {
                int idx = start + line_num - 1;
                ctx->marked[idx] = !ctx->marked[idx];

        }
        in_mark_mode = 0;
        mark_input_idx = 0;
    } else if (key == '\b' && mark_input_idx > 0) {
        mark_input_idx--;

        mark_input[mark_input_idx++] = key;
```
